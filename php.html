PHP Loops Assignment

PHP Code

    <?php
    
    // Question 1
    for ($i = 1; $i <= 18; $i++) {
     echo $i . " ";
    }
    
    // Question 2
    $i = 1;
    while ($i <= 27) {
     echo $i . " ";
     $i++;
    }
    
    // Question 3
    $i = 1;
    do {
     echo $i . " ";
     $i++;
    } while ($i <= 15);
    
    // Question 4
    $sum = 0;
    for ($i = 2; $i <= 21; $i++) {
     if ($i % 2 != 0) {
      $sum += $i;
     }
    }
    echo $sum;
    
    // Question 5
    $product = 1;
    for ($i = 3; $i <= 17; $i++) {
     if ($i % 2 == 0) {
      $product *= $i;
     }
    }
    echo $product;
    
    ?>
